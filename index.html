🎮 Il Mio Catalogo GOG

Visualizza la tua libreria di giochi in una galleria moderna e reattiva.

📋 Istruzioni per iniziare:

  1. Accedi al tuo account su GOG.com.
  2. Apri la console del browser (premi F12 oppure tasto destro -> Ispeziona e vai sulla scheda Console).
  3. Incolla lo script di estrazione JavaScript e premi Invio.
    4. (async function() { // 1. Creazione dell'interfaccia visiva (Overlay) su GOG const overlay = document.createElement('div'); overlay.style = "position:fixed;inset:0;background:rgba(17,17,17,0.9);z-index:99999;display:flex;flex-direction:column;justify-content:center;align-items:center;color:#fff;font-family:Arial,sans-serif;"; overlay.innerHTML = `

    🧙‍♂️ Estrattore Libreria GOG

    Inizializzazione...

    `; document.body.appendChild(overlay); const textEl = document.getElementById('gog-progress-text'); const barEl = document.getElementById('gog-progress-bar'); const games = []; async function fetchPage(pageNum) { const url = `/account/getFilteredProducts?hiddenFlag=0&mediaType=1&page=${pageNum}&sortBy=title&totalPages=1`; const resp = await fetch(url); return await resp.json(); } try { textEl.innerText = "Lettura della prima pagina..."; const firstPageData = await fetchPage(1); const totalPages = firstPageData.totalPages || 1; function extractFromData(data) { data.products.forEach(product => { if (product.isGame && (product.url || product.slug)) { const title = product.title; const link = "https://www.gog.com" + (product.url || `/en/game/${product.slug}`); let cover = product.image; if (cover && !cover.startsWith("http")) cover = "https:" + (cover.startsWith("//") ? cover : "//" + cover); if (cover && cover.includes(".jpg")) cover = cover.replace(".jpg", "_392.jpg"); games.push({ title, link, cover }); } }); } extractFromData(firstPageData); // Aggiorna progresso prima pagina barEl.style.width = `${(1 / totalPages) * 100}%`; textEl.innerText = `Analizzate 1 di ${totalPages} pagine...`; // Caricamento progressivo delle altre pagine per mostrare l'avanzamento reale for (let page = 2; page <= totalPages; page++) { const pageData = await fetchPage(page); extractFromData(pageData); // Calcolo percentuale barra const percent = (page / totalPages) * 100; barEl.style.width = `${percent}%`; textEl.innerText = `Analizzate ${page} di ${totalPages} pagine...`; } textEl.innerText = `Generazione file JSON (${games.length} giochi)...`; await new Promise(r => setTimeout(r, 500)); // Scarica il file JSON const blob = new Blob([JSON.stringify(games, null, 2)], { type: "application/json" }); const url = URL.createObjectURL(blob); const a = document.createElement("a"); a.href = url; a.download = "games_data.json"; document.body.appendChild(a); a.click(); setTimeout(() => { URL.revokeObjectURL(url); a.remove(); }, 1000); textEl.style.color = "#44ff44"; textEl.innerText = "📥 Download Completato!"; setTimeout(() => overlay.remove(), 1500); } catch (err) { textEl.style.color = "#ff5555"; textEl.innerText = "Errore durante l'estrazione!"; setTimeout(() => overlay.remove(), 3000); } })();
  4. Attendi il completamento e scarica il file games_data.json sul tuo computer.
